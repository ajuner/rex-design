内联模式
function MultiSelect() {
  const [selectedKeys, onSelect] = useState(['1', '3']);


  return (
    <div>
      <p>selectedKeys: {selectedKeys.join(',')}</p>
      <Menu
        style={{ width: 200 }}
        dataSource={[
          { key: '1', label: '选项 1 ', helper: 'Ctrl+P' },
          { key: '2', label: '选项 2' },
          { key: '3', label: 'Option 3' },
        ]}
        selectMode="multiple"
        selectedKeys={selectedKeys}
        onSelect={onSelect}
      />
    </div>
  );
}

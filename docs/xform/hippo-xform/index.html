export interface FormLayoutParams {
/** 标签位置，可选 'left' 或 'top' */
labelPosition?: 'left' | 'top';
// isInline?: boolean;
/** 标签宽度 */
labelWidth?: string | number;
/** 控件宽度 */
controlWidth?: string | number;
/** 两个 form item 之间的间距 */
formItemGap?: string | number;
}

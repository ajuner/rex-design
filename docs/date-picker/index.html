设置默认展示日期#

内联模式
function GetVisibleMonth() {
  return <DatePicker getDefaultVisibleMonth={() => dayjs('2020-12', 'YYYY-MM')} />;
}

自定义禁用日期#

内联模式
function DisabledDate() {
  const today = dayjs();


  return (
    <DatePicker
      getDisabledDate={(date) => {
        // 禁用今天之前的所有日期
        if (date.isBefore(today)) {
          return true;
        }
        return false;
      }}
    />
  );
}
